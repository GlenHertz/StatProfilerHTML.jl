91 function dsfmt_jump(s::DSFMT_state, jp::AbstractString)
92 index = s.val[end-1]
93 work = zeros(UInt64, JN32>>1)
94 dsfmt = reinterpret(UInt64, copy(s.val))
95 dsfmt[end] = UInt64(N*2)
96
97 for c in jp
98 bits = parse(UInt8,c,16)

110 function dsfmt_jump_add!(dest::Vector{UInt64}, src::Vector{UInt64})
111 dp = dest[end] >> 1
112 sp = src[end] >> 1
113 diff = ((sp - dp + N) % N)
114 i = 1
115 while i <= N-diff
116 j = i*2-1
117 p = j + diff*2
118 dest[j] $= src[p]
119 dest[j+1] $= src[p+1]
120 i += 1
121 end
122 while i <= N
123 j = i*2-1
124 p = j + (diff - N)*2
125 dest[j] $= src[p]
126 dest[j+1] $= src[p+1]
127 i += 1
128 end
129 dest[N*2+1] $= src[N*2+1]
130 dest[N*2+2] $= src[N*2+2]
131 return dest
132 end

134 function dsfmt_jump_next_state!(mts::Vector{UInt64})
135 POS1 = 117
136 SL1 = 19
137 SR = 12
138 MSK1 = 0x000ffafffffffb3f
139 MSK2 = 0x000ffdfffc90fffd
140
141 idx = (mts[end] >> 1) % N
142
143 a = idx*2+1
144 b = ((idx + POS1) % N)*2+1
145 u = N*2+1
146
147 t0 = mts[a]
148 t1 = mts[a+1]
149 L0 = mts[u]
150 L1 = mts[u+1]
151 mts[u] = (t0 << SL1) $ (L1 >> 32) $ (L1 << 32) $ mts[b]
152 mts[u+1] = (t1 << SL1) $ (L0 >> 32) $ (L0 << 32) $ mts[b+1]
153 mts[a] = (mts[u] >> SR) $ (mts[u] & MSK1) $ t0
154 mts[a+1] = (mts[u+1] >> SR) $ (mts[u+1] & MSK2) $ t1
155
156 mts[end] = (mts[end] + 2) % (N*2)
