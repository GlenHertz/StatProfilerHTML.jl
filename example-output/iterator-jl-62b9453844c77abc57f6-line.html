243 head_and_tail(c, n) -> head, tail
244
245 Returns `head`: the first `n` elements of `c`;
246 and `tail`: an iterator over the remaining elements.
247 """
248 function head_and_tail(c, n)
249 head = Vector{eltype(c)}(n)
250 s = start(c)
251 i = 0
252 while i < n && !done(c, s)
253 i += 1
254 head[i], s = next(c, s)
255 end

594 function start(f::Flatten)
595 local inner, s2
596 s = start(f.it)
597 d = done(f.it, s)
598 # this is a simple way to make this function type stable
599 d && throw(ArgumentError("argument to Flatten must contain at least one iterator"))
600 while !d
601 inner, s = next(f.it, s)
602 s2 = start(inner)
603 !done(inner, s2) && break

605 end
606 return s, inner, s2
607 end

613 inner, s = next(f.it, s)
614 s2 = start(inner)
615 end
616 return val, (s, inner, s2)
617 end

639
640
641 type PartitionIterator{T}
642 c::T
643 n::Int
644 end
645
646 eltype{T}(::Type{PartitionIterator{T}}) = Vector{eltype(T)}
647
648 function length(itr::PartitionIterator)
649 l = length(itr.c)
650 return div(l, itr.n) + ((mod(l, itr.n) > 0) ? 1 : 0)
651 end
652
653 start(itr::PartitionIterator) = start(itr.c)
654
655 done(itr::PartitionIterator, state) = done(itr.c, state)
656
657 function next{T<:Vector}(itr::PartitionIterator{T}, state)
658 l = state
659 r = min(state + itr.n-1, length(itr.c))
660 return view(itr.c, l:r), r + 1
661 end
662
663 function next(itr::PartitionIterator, state)
664 v = Vector{eltype(itr.c)}(itr.n)
665 i = 0
666 while !done(itr.c, state) && i < itr.n
667 i += 1
668 v[i], state = next(itr.c, state)
669 end
670 return resize!(v, i), state
671 end
