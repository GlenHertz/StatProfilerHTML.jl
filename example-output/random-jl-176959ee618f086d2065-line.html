515 m = last(r) - first(r)
516 m < 0 && throw(ArgumentError("range must be non-empty"))

1332 # Fill S (resized as needed) with a random subsequence of A, where
1333 # each element of A is included in S with independent probability p.
1334 # (Note that this is different from the problem of finding a random
1335 # size-m subset of A where m is fixed!)
1336 function randsubseq!(r::AbstractRNG, S::AbstractArray, A::AbstractArray, p::Real)
1337 0 <= p <= 1 || throw(ArgumentError("probability $p not in [0,1]"))
1338 n = length(A)
1339 p == 1 && return copy!(resize!(S, n), A)
1340 empty!(S)
1341 p == 0 && return S
1342 nexpected = p * length(A)
1343 sizehint!(S, round(Int,nexpected + 5*sqrt(nexpected)))
1344 if p > 0.15 # empirical threshold for trivial O(n) algorithm to be better
1345 for i = 1:n
1346 rand(r) <= p && push!(S, A[i])
1347 end
1348 else
1349 # Skip through A, in order, from each element i to the next element i+s
1350 # included in S. The probability that the next included element is
1351 # s==k (k > 0) is (1-p)^(k-1) * p, and hence the probability (CDF) that
1352 # s is in {1,...,k} is 1-(1-p)^k = F(k). Thus, we can draw the skip s
1353 # from this probability distribution via the discrete inverse-transform
1354 # method: s = ceil(F^{-1}(u)) where u = rand(), which is simply
1355 # s = ceil(log(rand()) / log1p(-p)).
1356 # -log(rand()) is an exponential variate, so can use randexp().
1357 L = -1 / log1p(-p) # L > 0
1358 i = 0
1359 while true
1360 s = randexp(r) * L
1361 s >= n - i && return S # compare before ceil to avoid overflow
1362 push!(S, A[i += ceil(Int,s)])
1363 end
1364 # [This algorithm is similar in spirit to, but much simpler than,
1365 # the one by Vitter for a related problem in "Faster methods for
1366 # random sampling," Comm. ACM Magazine 7, 703-718 (1984).]
1367 end
1368 return S
1369 end

1386 n = length(a)
1387 @assert n <= Int64(2)^52
1388 mask = nextpow2(n) - 1
1389 for i = n:-1:2
1390 (mask >> 1) == i && (mask >>= 1)
1391 j = 1 + rand_lt(r, i, mask)

1393 end
1394 return a
