335 function collect_to!{T}(dest::AbstractArray{T}, itr, offs, st)
336 # collect to dest array, checking the type of each result. if a result does not
337 # match, widen the result type and re-dispatch.
338 i = offs
339 while !done(itr, st)
340 el, st = next(itr, st)
341 S = typeof(el)
342 if S === T || S <: T
343 @inbounds dest[i] = el::T
344 i += 1
345 else
346 R = typejoin(T, S)
347 new = similar(dest, R)
348 copy!(new,1, dest,1, i-1)
349 @inbounds new[i] = el
350 return collect_to!(new, itr, i+1, st)
351 end
352 end
353 return dest
354 end

361 function grow_to!(dest, itr, st)
362 T = eltype(dest)
363 while !done(itr, st)
364 el, st = next(itr, st)
365 S = typeof(el)
366 if S === T || S <: T
367 push!(dest, el::T)
368 else
369 new = similar(dest, typejoin(T, S))
370 copy!(new, dest)
371 push!(new, el)

373 end

606 s = start(inds)
607 done(inds, s) && return a
608 (p, s) = next(inds, s)
609 q = p+1
610 while !done(inds, s)
611 (i,s) = next(inds, s)
612 if !(q <= i <= n)
613 if i < q
614 throw(ArgumentError("indices must be unique and sorted"))
615 else
616 throw(BoundsError())

619 while q < i

623 q = i+1

636 v = a[i]
637 m = length(ins)
638 if m == 0

640 elseif m == 1
641 a[i] = ins[1]
642 else
643 _growat!(a, i, m-1)

653 function splice!{T<:Integer}(a::Vector, r::UnitRange{T}, ins=_default_splice)
654 v = a[r]
655 m = length(ins)
656 if m == 0
657 deleteat!(a, r)
658 return v
659 end
660
661 n = length(a)
662 f = first(r)
663 l = last(r)
664 d = length(r)
665
666 if m < d
667 delta = d - m
668 _deleteat!(a, (f - 1 < n - l) ? f : (l - delta + 1), delta)
669 elseif m > d
670 _growat!(a, (f - 1 < n - l) ? f : (l + 1), m - d)
671 end
672
673 k = 1
674 for x in ins
675 a[f+k-1] = x
676 k += 1
677 end
678 return v
679 end

1449 """
1450 setdiff(a, b)
1451
1452 Construct the set of elements in `a` but not `b`. Maintains order with arrays. Note that
1453 both arguments must be collections, and both will be iterated over. In particular,
1454 `setdiff(set,element)` where `element` is a potential member of `set`, will not work in
1455 general.
1456
1457 ```jldoctest
1458 julia> setdiff([1,2,3],[3,4,5])
1459 2-element Array{Int64,1}:
1460 1
1461 2
1462 ```
1463 """
1464 function setdiff(a, b)
1465 args_type = promote_type(eltype(a), eltype(b))
1466 bset = Set(b)
1467 ret = Array{args_type,1}(0)
1468 seen = Set{eltype(a)}()
1469 for a_elem in a
1470 if !in(a_elem, seen) && !in(a_elem, bset)
1471 push!(ret, a_elem)
1472 push!(seen, a_elem)
1473 end
1474 end
1475 ret
1476 end
