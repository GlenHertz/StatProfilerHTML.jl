14 # binary GCD (aka Stein's) algorithm
15 # about 1.7x (2.1x) faster for random Int64s (Int128s)
16 function gcd{T<:Union{Int64,UInt64,Int128,UInt128}}(a::T, b::T)
17 a == 0 && return abs(b)
18 b == 0 && return abs(a)
19 za = trailing_zeros(a)
20 zb = trailing_zeros(b)
21 k = min(za, zb)
22 u = unsigned(abs(a >> za))
23 v = unsigned(abs(b >> zb))
24 while u != v
25 if u > v
26 u, v = v, u
27 end
28 v -= u
29 v >>= trailing_zeros(v)
30 end
31 r = u << k
32 # T(r) would throw InexactError; we want OverflowError instead
33 r > typemax(T) && throw(OverflowError())
34 r % T
35 end

51 """
52 gcdx(x,y)
53
54 Computes the greatest common (positive) divisor of `x` and `y` and their Bézout
55 coefficients, i.e. the integer coefficients `u` and `v` that satisfy
56 ``ux+vy = d = gcd(x,y)``. ``gcdx(x,y)`` returns ``(d,u,v)``.
57
58 ```jldoctest
59 julia> gcdx(12, 42)
60 (6,-3,1)
61 ```
62
63 ```jldoctest
64 julia> gcdx(240, 46)
65 (2,-9,47)
66 ```
67
68 !!! note
69 Bézout coefficients are *not* uniquely defined. `gcdx` returns the minimal
70 Bézout coefficients that are computed by the extended Euclidean algorithm.
71 (Ref: D. Knuth, TAoCP, 2/e, p. 325, Algorithm X.)
72 For signed integers, these coefficients `u` and `v` are minimal in
73 the sense that ``|u| < |y/d|`` and ``|v| < |x/d|``. Furthermore,
74 the signs of `u` and `v` are chosen so that `d` is positive.
75 For unsigned integers, the coefficients `u` and `v` might be near
76 their `typemax`, and the identity then holds only via the unsigned
77 integers' modulo arithmetic.
78 """
79 function gcdx{T<:Integer}(a::T, b::T)
80 # a0, b0 = a, b
81 s0, s1 = one(T), zero(T)
82 t0, t1 = s1, s0
83 # The loop invariant is: s0*a0 + t0*b0 == a
84 while b != 0
85 q = div(a, b)
86 a, b = b, rem(a, b)
87 s0, s1 = s1, s0 - q*s1
88 t0, t1 = t1, t0 - q*t1
89 end
90 a < 0 ? (-a, -s0, -t0) : (a, s0, t0)
91 end
92 gcdx(a::Integer, b::Integer) = gcdx(promote(a,b)...)

95 function invmod{T<:Integer}(n::T, m::T)
96 g, x, y = gcdx(n, m)
97 (g != 1 || m == 0) && throw(DomainError())

99 # For unsigned T, x might be close to typemax; add m to force a wrap-around.
100 r = mod(x + m, m)
101 # The postcondition is: mod(r * n, m) == mod(T(1), m) && div(r, m) == 0
102 r
103 end
