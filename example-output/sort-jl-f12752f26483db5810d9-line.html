602 issignleft(o::ForwardOrdering, x::Floats) = lt(o, x, zero(x))
603 issignleft(o::ReverseOrdering, x::Floats) = lt(o, x, -zero(x))
604 issignleft(o::Perm, i::Int) = issignleft(o.order, o.data[i])
605
606 function fpsort!(v::AbstractVector, a::Algorithm, o::Ordering)
607 i, j = lo, hi = nans2end!(v,o)
608 @inbounds while true
609 while i <= j && issignleft(o,v[i]); i += 1; end
610 while i <= j && !issignleft(o,v[j]); j -= 1; end
611 i <= j || break
612 v[i], v[j] = v[j], v[i]
613 i += 1; j -= 1
614 end
615 sort!(v, lo, j, a, left(o))
616 sort!(v, i, hi, a, right(o))
617 return v
618 end
