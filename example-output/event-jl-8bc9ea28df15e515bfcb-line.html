5 """
6 Condition()
7
8 Create an edge-triggered event source that tasks can wait for. Tasks that call `wait` on a
9 `Condition` are suspended and queued. Tasks are woken up when `notify` is later called on
10 the `Condition`. Edge triggering means that only tasks waiting at the time `notify` is
11 called can be woken up. For level-triggered notifications, you must keep extra state to keep
12 track of whether a notification has happened. The [`Channel`](:class:`Channel`) type does
13 this, and so can be used for level-triggered events.
14 """

34 """
35 notify(condition, val=nothing; all=true, error=false)
36
37 Wake up tasks waiting for a condition, passing them `val`. If `all` is `true` (the default),
38 all waiting tasks are woken, otherwise only one is. If `error` is `true`, the passed value
39 is raised as an exception in the woken tasks.

41 notify(c::Condition, arg::ANY=nothing; all=true, error=false) = notify(c, arg, all, error)
42 function notify(c::Condition, arg, all, error)
43 if all
44 for t in c.waitq
45 schedule(t, arg, error=error)
46 end
47 empty!(c.waitq)
48 elseif !isempty(c.waitq)
49 t = shift!(c.waitq)
50 schedule(t, arg, error=error)
51 end
52 nothing
53 end

76 function enq_work(t::Task)
77 t.state == :runnable || error("schedule: Task not runnable")
78 ccall(:uv_stop, Void, (Ptr{Void},), eventloop())
79 push!(Workqueue, t)
80 t.state = :queued

86 """
87 schedule(t::Task, [val]; error=false)
88
89 Add a task to the scheduler's queue. This causes the task to run constantly when the system
90 is otherwise idle, unless the task performs a blocking operation such as `wait`.
91
92 If a second argument `val` is provided, it will be passed to the task (via the return value of
93 `yieldto`) when it runs again. If `error` is `true`, the value is raised as an exception in
94 the woken task.
95 """
96 function schedule(t::Task, arg; error=false)
97 # schedule a task to be (re)started with the given value or exception
98 if error
99 t.exception = arg
100 else
101 t.result = arg
102 end
103 return enq_work(t)
104 end

107 function schedule_and_wait(t::Task, v=nothing)
108 t.state == :runnable || error("schedule: Task not runnable")
109 if isempty(Workqueue)

111 else

128 """
129 yieldto(task, arg = nothing)
130
131 Switch to the given task. The first time a task is switched to, the task's function is
132 called with no arguments. On subsequent switches, `arg` is returned from the task's last
133 call to `yieldto`. This is a low-level call that only switches tasks, not considering states
134 or scheduling in any way. Its use is discouraged.
135 """
136 yieldto(t::Task, x::ANY = nothing) = ccall(:jl_switchto, Any, (Any, Any), t, x)

144 function wait()
145 while true
146 if isempty(Workqueue)
147 c = process_events(true)
148 if c==0 && eventloop()!=C_NULL && isempty(Workqueue)
149 # if there are no active handles and no runnable tasks, just
150 # wait for signals.
151 pause()
152 end
153 else
154 t = shift!(Workqueue)
155 if t.state != :queued
156 # assume this somehow got queued twice,
157 # probably broken now, but try discarding this switch and keep going
158 # can't throw here, because it's probably not the fault of the caller to wait
159 # and don't want to use print() here, because that may try to incur a task switch
160 ccall(:jl_safe_printf, Void, (Ptr{UInt8}, Vararg{Int32}),
161 "\nWARNING: Workqueue inconsistency detected: shift!(Workqueue).state != :queued\n")
162 continue
163 end
164 arg = t.result
165 t.result = nothing
166 t.state = :runnable
167 local result
168 try
169 result = yieldto(t, arg)
170 current_task().state == :runnable || throw(AssertionError("current_task().state == :runnable"))
171 catch e
172 ct = current_task()
173 if ct.state == :queued
174 if t.state == :runnable
175 # assume we failed to queue t
176 # return it to the queue to be scheduled later
177 t.result = arg
178 t.state = :queued
179 push!(Workqueue, t)
180 end
181 # return ourself to the runnable state
182 i = findfirst(Workqueue, ct)
183 i == 0 || deleteat!(Workqueue, i)
184 ct.state = :runnable
185 end
186 rethrow(e)
187 end
188 process_events(false)
189 # return when we come out of the queue
190 return result
191 end
192 end
193 assert(false)
194 end

196 if is_windows()
197 pause() = ccall(:Sleep, stdcall, Void, (UInt32,), 0xffffffff)
198 else
199 pause() = ccall(:pause, Void, ())
200 end
